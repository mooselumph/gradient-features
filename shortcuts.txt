
# dockerfile --> image

docker build 

# image --> container

docker create

# image --> build container and start

docker run --gpus all -d -t --name CONTAINER_NAME --mount type=bind,source="$(pwd)"/code/jax,target=/app IMAGE_ID
docker run --gpus all -d -t --name CONTAINER_NAME --mount type=bind,source="$(pwd)"/code/jax,target=/app REPOSITORY:VERSION (e.g jax:latest)

# Start and stop container

docker stop CONTAINER_NAME
docker start CONTAINER_NAME

# Login to running container

docker exec -it CONTAINER_NAME /bin/bash

# Jupyter / CUDA images

docker build --tag nvidia-jupyter --build-arg ROOT_CONTAINER=nvidia/cuda:10.1-cudnn7-devel https://github.com/jupyter/docker-stacks.git#master:base-notebook

ssh -L 8888:localhost:8888 USER@hilbert.ee.washington.edu
docker run --gpus all -p 8888:8888 jax-jupyter
